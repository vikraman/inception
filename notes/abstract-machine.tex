\subsubsection{Grammar}

\begin{align*}
  \text{Types:} && X \,,\, Y \,,\, Z  &:=\; \unit \mid X \Rightarrow Y \mid X \times Y \mid \LABEL\\
  \text{Contexts:} && \Gamma \,,\, \Delta &:=\; \emptyctx \mid \Gamma , X\\
  \text{Variables:} && \ell \,,\, x \,,\, y &:=\; h \mid t ( x )
\end{align*}

The letter $\ell$ will be used to denote variables of type $\LABEL$ exclusively, while the letters $x$ and $y$ will be used for all variables.

Furthermore, we let $R_0 = V$ (this is a type) and $R = \VP$ (this is a presheaf) for the rest of the document.
We also fix a continuation $k_0 : \llbracket R_0 \rrbracket \to \llbracket R \rrbracket$ (interpretation is given below).

\subsubsection{Typing rules for membership}

   \frac{}{h : X \in \Gamma , X}
  && \frac{x : X \in \Gamma}{t(x) : X \in \Gamma , Y}
\end{align*}

\subsubsection{Interpretation of types for monad T (where $T = S$ or $T = K_R$)}$ $

Types are interpreted as presheaves in $\textsf{Set}^\mathbb{F}$.
\begin{align*}
  {\llbracket \unit \rrbracket}_T &= \top\\
  {\llbracket X \times Y \rrbracket}_T &= {\llbracket X \rrbracket}_T \times {\llbracket Y \rrbracket}_T\\
  {\llbracket X \Rightarrow Y \rrbracket}_T &= {\llbracket X \rrbracket}_T \Rightarrow T{\llbracket Y \rrbracket}_T\\
  {\llbracket \LABEL \rrbracket}_T &= R
\end{align*}

If $T = K_R$, we will just write $\llbracket X \rrbracket$ instead of ${\llbracket X \rrbracket}_{K_R}$.\newline

\subsubsection{Interpretation of contexts for monad T (where $T = S$ or $T = K_R$)}

\begin{align*}
  {\llbracket \emptyctx \rrbracket}_T &= \top\\
  {\llbracket \Gamma, X \rrbracket}_T &= {\llbracket \Gamma \rrbracket}_T \times {\llbracket X \rrbracket}_T
\end{align*}

If $T = K_R$, we will just write $\llbracket \Gamma \rrbracket$ instead of ${\llbracket \Gamma \rrbracket}_{K_R}$.\newline

\subsubsection{Interpretation of variables}$ $

Variables $x : X \in \Gamma$ are interpreted as morphisms $\llbracket \Gamma \rrbracket \xrightarrow{\llbracket x \rrbracket} \llbracket X \rrbracket$ in $\textsf{Set}^\mathbb{F}$.

\begin{align*}
  {\llbracket h \rrbracket} &= \pi_2\\
  {\llbracket t(x) \rrbracket} &= {\llbracket x \rrbracket}_T \circ \pi_1
\end{align*}

\begin{align*}

  \text{Evaluated value terms:} && \underline{M} \,,\, \underline{N} &:=\; \underline{\ast} \mid (\underline{M} \mid \underline{N}) \mid \underline{\lambda} x . W \mid \underline{\ell} \;\text{(where $\ell$ is a label variable)}\\

  \text{Value terms:} && \mathbf{M} \,,\, \mathbf{N} &:=\; M \mid \underline{M} \mid \overline{M}\\
  \text{Full computation terms:} && W &:=\; \RET (M) \mid \PUSH (M , N) \mid \APP (M , N) \mid \\

\subsubsection{Interpretation of full value terms for monad T (where $T = S$ or $T = K_R$)}$ $

Full value terms $\Gamma \VTS M : X$ are interpreted as morphisms ${\llbracket \Gamma \rrbracket}_T \xrightarrow{{\llbracket M \rrbracket}_T} {\llbracket X \rrbracket}_T$.

\begin{align*}
  {\llbracket \Gamma \VTS x : X \rrbracket}_T &= {\llbracket x \rrbracket}_T \\
  {\llbracket \Gamma \VTS \lambda W : X \Rightarrow Y \rrbracket}_T &= \lambda {\llbracket W \rrbracket}_T \\
  {\llbracket \Gamma \VTS (M, N) : X \times Y \rrbracket}_T &= \langle {\llbracket M \rrbracket}_T, {\llbracket N \rrbracket}_T \rangle \\
  {\llbracket \Gamma \VTS \PMV(M , N) : X \rrbracket}_T &= {\llbracket N \rrbracket}_T \circ \alpha \circ \langle \ID , {\llbracket M \rrbracket}_T \rangle \\
  {\llbracket \Gamma \VTS \ast : \unit \rrbracket}_T &= \top
\end{align*}

If $T = K_R$, we will just write $\llbracket \Gamma \VTS M : X \rrbracket$ instead of ${\llbracket \Gamma \VTS M : X \rrbracket}_{K_R}$.\newline


\subsubsection{Interpretation of full computation terms for monad $S$}$ $

Computation terms $\Gamma \CTS W : X$ are interpreted by ${\llbracket \_ \rrbracket}_S$ as morphisms ${\llbracket \Gamma \rrbracket}_S \xrightarrow{{\llbracket W \rrbracket}_S} S\llbracket X\rrbracket$.

\begin{align*}
  {\llbracket \Gamma \CTS \RET (M) : X \rrbracket}_S &= \eta \circ {\llbracket M \rrbracket}_S \\
  {\llbracket \Gamma \CTS \PUSH (W_1, x.W_2) : X\rrbracket}_S &= {\llbracket W_2 \rrbracket}_S^{\sharp} \circ \STR \circ \langle \ID , {\llbracket W_1 \rrbracket}_S \rangle\\
  {\llbracket \Gamma \CTS \APP(M , N) : X \rrbracket}_S &= \EVAL \circ \langle {\llbracket M \rrbracket}_S , {\llbracket N \rrbracket}_S \rangle\\
  {\llbracket \Gamma \CTS \VAR(M) : X \rrbracket}_S &= \VAR_X \circ {\llbracket M \rrbracket}_S\\
  {\llbracket \Gamma \CTS \SUB(a.W_1 , W_2) : X \rrbracket}_S &= \SUB_X \circ \langle \lambda {\llbracket W_1 \rrbracket}_S , {\llbracket W_2 \rrbracket}_S \rangle\\
  {\llbracket \Gamma \CTS \PMC(M , (x , y).W) : X \rrbracket}_S &= {\llbracket W \rrbracket}_S \circ \alpha \circ \langle \ID , \llbracket M \rrbracket \rangle
\end{align*}

\subsubsection{Interpretation of full computation terms for monad $K_R$}$ $

Computation terms $\Gamma \CTS W : X$ are interpreted by $\llbracket \_ \rrbracket$ as morphisms $\llbracket \Gamma \rrbracket \xrightarrow{\llbracket W \rrbracket} K_R\llbracket X\rrbracket$.

\begin{align*}
  \llbracket \Gamma \CTS \RET (M) : X \rrbracket \;\gamma\; k &= k \;(\llbracket M \rrbracket\; \gamma)\\
  \llbracket \Gamma \CTS \PUSH (W_1, x.W_2) : X\rrbracket \;\gamma \; k &= ({\llbracket W_2 \rrbracket}^{\sharp} \circ \STR \circ \langle \gamma , \llbracket W_1 \rrbracket \;\gamma \rangle) \; k\\
  \llbracket \Gamma \CTS \APP(M , N) : X \rrbracket \; \gamma \; k &= (\EVAL \circ \langle \llbracket M \rrbracket , \llbracket N \rrbracket \rangle) \; \gamma \; k\\
  \llbracket \Gamma \CTS \VAR(L) : X \rrbracket \; \gamma \; k &= \llbracket L \rrbracket \; \gamma\\
  \llbracket \Gamma \CTS \SUB(a.W_1 , W_2) : X \rrbracket \; \gamma \; k &= \llbracket W_1 \rrbracket \; (\gamma , \llbracket W_2 \rrbracket \, \gamma \, k) \; k\\
  \llbracket \Gamma \CTS \PMC(M , (x , y).W) : X \rrbracket \; \gamma \; k &= (\llbracket W \rrbracket \circ \alpha \circ \langle \gamma , \llbracket M \rrbracket \rangle) k
\end{align*}

\subsubsection{Typing rules for environments}
\begin{align*}
  \frac{}{\emptyenv : \Env{\emptyctx}}
  && \frac{\gamma : \Env{\Gamma} \quad \Gamma \VVTS \underline{M} : X }{\gamma , \underline{M} : \Env{\Gamma , X}}
  && \frac{\gamma : \Env{\Gamma} \quad \Gamma \CTS W : X \quad C : \CStack{\Delta}{X} \quad \Gamma = \Delta, \Delta'}{\gamma , \cspair{W}{C} : \Env{\Gamma , \LABEL}}
\end{align*}

\subsubsection{Interpretation of computation stacks}$ $

Computation stacks for any context and type $X$ are interpreted as continuations $\llbracket X \rrbracket \to R$.

In order to define this interpretation inductively, we first set up an intermediate interpretation ${\llbracket \_ \rrbracket}_K$, mapping computation stacks $C$ for any context and type $X$ to morphisms $K_R\llbracket X \rrbracket \xrightarrow{{\llbracket C \rrbracket}_K} K_{\LABEL}\llbracket R_0 \rrbracket$.
%
\begin{align*}
  {\llbracket \emptycs \rrbracket}_K &= \ID\\
  {\llbracket W \triangleleft \gamma :: C \rrbracket}_K &= {\llbracket C \rrbracket}_K \circ {\llbracket W \rrbracket}^{\sharp} \circ \STR \circ \langle \llbracket \gamma \rrbracket , \ID \rangle
\end{align*}

Using the starting continuation $k_0 : \llbracket R_0 \rrbracket \to R$, we can now interpret computation stacks $C$ for any context and for type $X$ as continuations $\llbracket X \rrbracket \to R$:
\[
  {\llbracket C \rrbracket} (x) = \big({\llbracket C \rrbracket}_K \; (\, \eta (x) \,) \big) \; k_0
\]

\subsubsection{Interpretation of environments}$ $

Environments $\gamma$ for context $\Gamma$ are interpreted as morphisms $\top \to \llbracket \Gamma \rrbracket$.
\begin{enumerate}
  \item $\llbracket () \rrbracket = \langle \rangle$
  \item $\llbracket (\gamma , M) \rrbracket = (\llbracket \gamma \rrbracket , \llbracket M \rrbracket \llbracket \gamma \rrbracket)$
  \item $\llbracket (\gamma , (W ; c)) \rrbracket = (\llbracket \gamma \rrbracket , \llbracket W \rrbracket \llbracket \gamma \rrbracket \llbracket c \rrbracket)$
\end{enumerate}

\subsection{Termination of the Machine}

To prove termination of the computation machine, we associate a natural number with each state and then show that this number strictly decreases when taking a transition.

We compute this number inductively for computation terms and value terms. In order to do so we need a type-indexed data structure that holds intermediate results for previously computed sub-terms.
We call this data structure a \emph{metric}.

\newcommand{\EMetric}{\ensuremath{\mathop{\textsf{EMetric}}}}

\begin{definition}
$\TermMetric X$ is an inductively defined type with the following constructors:

\hspace{20pt}$\begin{aligned}
	\mU &: (n : \bN) \to \TermMetric 1\\
	\mV &: (n : \bN) \to (w : \bN) \to \TermMetric V\\
	\mF &: (n : \bN) \to (c : \bN) \to (m : \TermMetric Y) \to \TermMetric (X \Rightarrow Y)\\
	\mP &: (n : \bN) \to (m_1 : \TermMetric X) \to (m_2 : \TermMetric Y) \to \TermMetric (X \times Y)\\
\end{aligned}$
\end{definition}

The parameter $n$ of each of the constructors is used to hold the number that a term adds to the metrics of its sub-terms.
The parameter $w$ is the number associated with the computation that a label variable is pointing to.
The parameter $c$ is a counter, giving an upper bound to the number of occurrences of the bound variable in a term of function type \emph{after} its evaluation.
We also define the following `projection' functions $\# : \TermMetric (X \Rightarrow Y) \to \bN$ (extracting the counter of a function metric), $\wpart : \TermMetric (X \Rightarrow Y) \to \TermMetric Y$ (extracting the $w$ part of a function metric), $\lhs : \TermMetric (X \Rightarrow Y) \to \TermMetric X$ (extracting the left-hand side metric), $\rhs : \TermMetric (X \Rightarrow Y) \to \TermMetric X$ (extracting the right-hand side metric), $\val : \TermMetric X \to \bN$ (extracting the $n$ part).
%
To add a natural number $n_1$ to the $n$ part of a metric $m : \TermMetric X$, we write $n_1 + m$, which is defined in the obvious way.

Every term exists in the context of an environment and a computation stack, the latter of which is (see Section~\ref{sec:env}) a list of computation terms (in an environment) with a free variable.


\begin{definition}\label{def:evl}
The function $\evl{\_} : \CStack \Delta X \to \List (\bN \times \bN)$ associates with each computation stack a list of natural numbers (called \emph{c-lists}), where:
\begin{itemize}
	\item The first component is an upper bound to the number of occurrences of the free variable (after evaluation) of the computation term.
  \item The second component is the number associated with the computation term.
\end{itemize}
\end{definition}

Once we have associated a number $n$ to a term $W$ in an environment on a stack $C$, we can calculate the number ${\| n \|}(l)$ associated to $C$ (where $l = \evl{C}$) with $W$ on the top.
We do this inductively as follows:
\begin{itemize}
	\item $\| n \| ([]) = 0$
	\item $\| n \| ((c , w) \CAT l) = w + n \cdot (1 + c) + \| w + n \cdot (1 + c) \| (l)$
\end{itemize}

We define partial orders on metrics and on c-lists.

\begin{definition}
We define the relation $\leq^M_X \,\subseteq \TermMetric X \times \TermMetric X$ inductively as follows (we will usually omit the subscript):

\hspace{10pt}$\begin{aligned}
	(\mU \, n_1) &\leq^M (\mU \, n_2) && \quad\quad\textsf{if $n_1 \leq n_2$}\\
	(\mV \, n_1 \, w_1) &\leq^M (\mV \, n_2 \, w_2) && \quad\quad\textsf{if $n_1 \leq n_2$ and $w_1 \leq w_2$}\\
	(\mF \, n_1 \, c \, m_1) &\leq^M (\mF \, n_2 \, c \, m_2) && \quad\quad\textsf{if $n_1 \leq n_2$ and $m_1 \leq^M m_2$}\\
	(\mP \, n_1 \, m_{1,1} \, m_{2,1}) &\leq^M (\mF \, n_2 \, m_{1,2} \, m_{2,2}) && \quad\quad\textsf{if $n_1 \leq n_2$ and $m_{1,1} \leq^M m_{1,2}$ and $m_{2,1} \leq^M m_{2,2}$}
\end{aligned}$
\end{definition}

\begin{definition}
We define the relation $\leq^c \,\subseteq (\List (\bN \times \bN) \times \List (\bN \times \bN))$ inductively as follows:

\hspace{10pt}$\begin{aligned}
	l &\leq^c l && \quad\quad\textsf{if $n_1 \leq n_2$}\\
	(c , w_1) \CAT l_1 &\leq^c (c , w_2) \CAT l_2 && \quad\quad\textsf{if $w_1 \leq w_2$ and $l_1 \leq^c l_2$}
\end{aligned}$
\end{definition}

Every term-environment pair in an environment is associated with a monotone function from c-lists to metrics $\List (\Sigma_{X : \textsf{Type}} (\List (\bN \times \bN) \xrightarrow{\leq} \TermMetric X))$.
To simplify notation we will write $\EElem X$ for $(\List (\bN \times \bN) \xrightarrow{\leq} \TermMetric X)$ and $\EMetric$ for $\List (\Sigma_{X : \textsf{Type}} \EElem X)$ from now on.

\end{definition}

Before we can calculate a metric for a term, we need to relate $\EMetric$'s to contexts, so ensure that an $\EMetric$ indeed corresponds to a given environment in that context.
The idea is that if a type appears in a context at a given position, then we need to know if there is a corresponding entry in the $\EMetric$ (which is a list) and if so, which one.
This means, that a context $\Gamma : \Ctx$ is ``weaker'' than a corresponding $E : \EMetric$.

\newcommand{\Wkn}{\ensuremath{\mathop{\textsf{Wkn}}}}
\newcommand{\wknnil}{\ensuremath{\mathop{\textsf{wkn-nil}}}}
\newcommand{\wkncong}{\ensuremath{\mathop{\textsf{wkn-cong}}}}
\newcommand{\wknwk}{\ensuremath{\mathop{\textsf{wkn-wk}}}}

\begin{definition}
We define the relation $\Wkn \subseteq \Ctx \times \EMetric$ inductively as follows:

\hspace{20pt}$\begin{aligned}
	\wknnil &: \Wkn\, \varepsilon\, []\\
	\wkncong &: (e : \EElem\, X) \to (\varpi : \Wkn\, \Gamma\, E) \to \Wkn\, (\Gamma , X)\, ((X , e) \CAT E)\\
	\wknwk &: (\varpi : \Wkn\, \Gamma\, E) \to \Wkn\, (\Gamma , X)\, E\\
\end{aligned}$
\end{definition}

\newcommand{\lookupeelem}{\ensuremath{\mathop{\textsf{lookup-eelem}}}}
\newcommand{\zm}{\ensuremath{\mathop{\textsf{zero-metric}}}}
\newcommand{\ze}{\ensuremath{\mathop{\textsf{zero-eelem}}}}

For every type $X$, there exists a $\zm : \TermMetric X$, constructed with zeros as parameters to all its constructors.
We also define a corresponding $\ze : \EElem X$ as the constant function to the $\zm$.
From a given metric $m : \TermMetric X$ we can calculate the corresponding natural number with the following function.

\newcommand{\mton}[1]{\ensuremath{\langle\!\langle \,#1\, \rangle\!\rangle}}

\begin{definition}
$\mton{\_} : \TermMetric X \to \bN$ is inductively defined as follows:

\hspace{20pt}$\begin{aligned}
	\mton{\mU\, n}       &= n\\
	\mton{\mV\, n\, w}       &= n + w\\
	\mton{\mF\, n\, c\, m}       &= n + \mton{m}\\
	\mton{\mP\, n\, m_1\, m_2}       &= n + \mton{m_1} + \mton{m_2}\\
\end{aligned}$

\end{definition}


We can now define a function to lookup the $\EElem X$ for a variable of type $X$ in an $\EMetric$.


\begin{definition}
The function
\[\lookupeelem : (i : \Gamma \ni X) \to (E : \EMetric) \to (\varpi : \Wkn\, \Gamma\, E) \to \EElem X\]
is inductively defined as follows:

\hspace{20pt}$\begin{aligned}
	\lookupeelem\, h\, ((X , e) \CAT E)\, (\wkncong\, e\, \varpi)\,       &= e\\
	\lookupeelem\, (t\, i)\, ((X , e) \CAT E)\, (\wkncong\, e\, \varpi)\,   &= \lookupeelem \,i \,E \,\varpi\\
	\lookupeelem\, h\, []\, (\wknwk \,\varpi)\,       &= \ze\\
	\lookupeelem\, h\, ((X , e) \CAT E)\, (\wknwk \varpi)\,       &= \ze\\
	\lookupeelem\, (t\, i)\, []\, (\wknwk \varpi)\,       &= \ze\\
	\lookupeelem\, (t\, i)\, ((X , e) \CAT E)\, (\wknwk \varpi)\,       &= \lookupeelem \,i \,((X , e) \CAT E) \,\varpi\\
\end{aligned}$

\end{definition}

We can now give a precise definition for the function
\[ \evl{\_} : \CStack \Delta X \to \List (\bN \times \bN) \]
introduced in Definition \ref{def:evl} and also define a function to calculate an $\EMetric$ together with its correctness property from an environment.

\begin{definition}\label{def:env}
$\evl{\_} : \Env{\Gamma} \to \Sigma_{E : \EMetric} \Wkn\, \Gamma\, E$ calculates from an environment $\gamma : \Env{\Gamma}$ a corresponding pair $(E , \Wkn\, \Gamma\, E)$ and is defined by mutual induction with the function
\[ \evl{\_} : \CStack \Delta X \to \List (\bN \times \bN) \]
from Definition \ref{def:evl}:

\hspace{20pt}$\begin{aligned}
 \evl{ \emptycs } & = [] \\
 \evl{ \cstack{W}{\gamma}{C} } & = ((\ccount\, h\, W\, (\pi_1\evl{\gamma})\, (\pi_2\evl{\gamma}))\, \evl{C} ,\,\\
                               &\quad\quad\quad \mton{(\ceelem\, W\, (\pi_1\evl{\gamma})\, (\wknwk \pi_2\evl{\gamma}))\, \evl{C}} )\CAT \evl{C} \\
 \evl{ \emptyenv } &= ([] , \wknnil)\\
 \evl{ \gamma , M } &= ((\_, \underline{\veelem}\, M\, (\pi_1\evl{\gamma})\, (\pi_2\evl{\gamma}) ) \CAT (\pi_1\evl{\gamma}) , (\wkncong \, \_ \, (\pi_2\evl{\gamma})) )\\
 \evl{ \gamma , \cspair{W}{C} } &= ((\_, y \mapsto \mV 0 (\mton{(\ceelem\, M\, (\pi_1\evl{\gamma})\, (\pi_2\evl{\gamma}))\, \evl{C}} \\
                                & \quad\quad\quad\quad           + \|\underline{\veelem}\, M\, (\pi_1\evl{\gamma})\, (\pi_2\evl{\gamma}) )\|(\evl{C})    ) ) \CAT (\pi_1\evl{\gamma}) ,\\
                                & \quad\quad\quad (\wkncong \, \_ \, (\pi_2\evl{\gamma})) )\\
\end{aligned}$

\end{definition}

We can now calculate $\EElem$'s (functions from c-lists to metrics) for the states of the lookup machine and the value machine as follows:

\hspace{20pt}$\begin{aligned}
|\LS{x}{\gamma}| \;&= \; \lookupeelem\, x\, (\pi_1\, \evl{\gamma})\, (\pi_2\, \evl{\gamma})\\
|\VS{\dots \CAT \vstack{M}{\gamma}{\emptyvs}}| \;&=\; \veelem\, M\, (\pi_1\, \evl{\gamma})\, (\pi_2\, \evl{\gamma})\\
|\VS{\dots \CAT \vstack{\underline{M}}{\gamma}{\emptyvs}}| \;&=\; \underline{\veelem}\, \underline{M}\, (\pi_1\, \evl{\gamma})\, (\pi_2\, \evl{\gamma})\\
\end{aligned}$

For computation states we can calculate the associated natural number as follows:

\hspace{20pt}$\begin{aligned}
|\CSA{W}{\gamma}{C}| \;&=\; \mton{(\ceelem\, W\, (\pi_1\, \evl{\gamma})\, (\pi_2\, \evl{\gamma}))\,\evl{C}}\\
|\CSB{\underline{W}}{\gamma}{C}| \;&=\; \mton{(\underline{\ceelem}\, \underline{W}\, (\pi_1\, \evl{\gamma})\, (\pi_2\, \evl{\gamma}))\,\evl{C}}
\end{aligned}$

\vspace{20pt}
I have proven the following theorems:

\begin{theorem}\label{thm:ldecr}
For all c-lists $l : \List\, (\bN \times \bN)$, and all lookup states $\LS{x}{\gamma}$, if there exists a sequence of transitions $\LS{x}{\gamma} \leadsto^{L*} L_H$, where $L_H$ is a halting lookup state, then:
\[ \mton{|\LS{x}{\gamma}| \, l} \leq \mton{|L_T| \, l} \]
\end{theorem}\label{thm:vdecr}

\begin{theorem}
For all c-lists $l : \List\, (\bN \times \bN)$, and all value states $\VS{\vstack{M}{\gamma}{\emptyvs}}$, if there exists a sequence of transitions $\VS{\vstack{M}{\gamma}{\emptyvs}} \leadsto^{V*} V_H$, where $V_H$ is a halting value state, then:
\[ \mton{|\VS{\vstack{M}{\gamma}{\emptyvs}}| \, l} \leq \mton{|V_T| \, l} \]
\end{theorem}

\vspace{20pt}
The proofs of these theorems depends on a multitude of further definitions and lemmas, relating `weakenings' $\Wkn\, \Gamma\, E$ and $\Wkn\, \Gamma'\, E'$, when $\Gamma$ is a weaker context than $\Gamma'$ in the usual sense.

However, with the definitions as given above, the metric does \emph{not} decrease for computation states.
The issue lies in how the upper bound for occurrences of variables in a term is calculated.
I believe that this can be fixed easily by adjusting the $\APP$ case in the definition of $\ceelem$, but this will require reproving all lemmas on which the proofs of Theorems~\ref{thm:ldecr} and \ref{thm:vdecr} depend.
