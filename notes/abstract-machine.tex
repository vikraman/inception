\subsection{Terms}
We have value terms and computation terms.
For the purposes of the abstract machine, we distinguish between full value terms, partial value terms and evaluated value terms.
We also distinguish between full computation terms and evaluated computation terms.
Every full value term is also a partial value term.
Every evaluated value term is also a partial value term.


Full value terms are:

\quad
$x$,\,
$\ast$,\,
$\lambda x . W$,\,
$(L , R)$,\,
$\PMV ( M , (x,y).N )$

Evaluated value terms are:

\quad
$\underline{\ell}$ (where $\ell$ is a label variable),\,
$\underline{\ast}$,\,
$(\underline{M}, \underline{N})$,\,
$\underline{\lambda} x . W$

Further partial value terms are:

\quad
$(\_ , R)$,\,
$(L , \_)$,\,
$\flag{(L , R)}$,\,
$\PMV(\_, (x , y).N)$

Full computation terms are:

\quad
$\RET (M)$,\,
$\PUSH (M , x.N)$,\,
$\APP (M , N)$,\,
$\VAR (M)$,\,
$\SUB (a.M, N)$,\,
$\PMC ( M , (x,y).N )$

Evaluated computation terms are:

\quad
$\underline{\APP} (\underline{M} , \underline{N})$,\,
$\underline{\RET} (\underline{M})$,\


Typing rules for values:

\begin{align*}
   \frac{x : X \in \Gamma}{\Gamma \VTS x : X}
&& \frac{\Gamma , x : X \CTS M : Y}{\Gamma \VTS \lambda x . M : X \Rightarrow Y}
&& \frac{\Gamma \VTS M : X \quad \Gamma \VTS N : Y}{\Gamma \VTS (M , N) : X \times Y}
\end{align*}
\begin{align*}
\frac{\Gamma \VTS M : X \times Y \quad \Gamma, x : X , y : Y \VTS N : Z}{\Gamma \VTS \PMV(M, (x , y).N) : Z}
&& \frac{}{\Gamma \VTS \ast : \unit}
\end{align*}

Typing rules for computations:

\begin{align*}
   \frac{\Gamma \VTS M : X}{\Gamma \CTS \RET (M) : X}
&& \frac{\Gamma \CTS M : X \quad \Gamma, x : X \CTS N : Y}{\Gamma \CTS \PUSH (M , x.N) : Y}
&& \frac{\Gamma \VTS M : X \Rightarrow Y \quad \Gamma \VTS N : X}{\Gamma \CTS \APP(M, N) : Y}
\end{align*}
\begin{align*}
   \frac{\Gamma \VTS M : \LABEL}{\Gamma \CTS \VAR_X (M) : X}
&& \frac{\Gamma , a : \LABEL \CTS M : X \quad \Gamma \CTS N : X}{\Gamma \CTS \SUB (a.M , N) : X}
&& \frac{\Gamma \VTS M : X \times Y \quad \Gamma, x : X , y : Y \CTS N : Z}{\Gamma \CTS \PMC(M, (x,y).N) : Z}
\end{align*}

The typing rules for evaluated value and computation terms are similar and we will use the same notation for both typing judgements.

\subsection{Environments}

We fix a type $R_0$.
%
The unique environment for the empty context is an empty list denoted by $()$.
An environment $\gamma$ for context $\Gamma, \Gamma', x : X$ is either:
\begin{itemize}
	\item A pair $(\gamma' , \underline{M})$, where $\Gamma, \Gamma' \VTS \underline{M} : X$ is an evaluated value term and $\gamma'$ is an environment for context $\Gamma, \Gamma'$, or
  \item A pair $(\gamma' , (W ; c))$, where $\Gamma, \Gamma' \CTS W : X$ is a full computation term, $c$ is a computation stack for context $\Gamma$ and type $X$ and $\gamma'$ is an environment for context $\Gamma, \Gamma'$.
\end{itemize}

The unique computation stack for the empty context and type $R_0$ is an empty list denoted by $\square$.
A computation stack for context $\Gamma$ and type $Z$ is a triple $W \triangleleft \gamma :: c$, where $\Gamma , \Gamma' , z : Z \CTS W : X$ is a computation term, $\gamma$ is an environment for $\Gamma , \Gamma'$ and $c$ is a computation stack for $\Gamma$ and $X$.
%We will use the notation $W \triangleleft \gamma :: c$ to denote these triples.

\subsection{Interpretation}
Types $A$ are interpreted as presheaves $\llbracket A \rrbracket$ and contexts $\Gamma = x_1 : A_1 , \dots, x_n : A_n$ as products of presheaves $\llbracket \Gamma \rrbracket = \llbracket A_1 \rrbracket \times \dots \times \llbracket A_n \rrbracket$.
The empty context is interpreted as the terminal object 1.

Value terms $\Gamma \VTS M : A$ are interpreted as morphisms $\llbracket \Gamma \rrbracket \xrightarrow{\llbracket M \rrbracket} \llbracket A\rrbracket$.
\begin{enumerate}
\item $\llbracket x_1 : A_1 , \dots, x_n : A_n \VTS x_i : A_i \rrbracket = \pi_i$
\item $\llbracket \Gamma \VTS \lambda x.M : A \Rightarrow B \rrbracket = \lambda \llbracket M \rrbracket$
\item $\llbracket \Gamma \VTS (M, N) : A \times B \rrbracket = \langle \llbracket M \rrbracket, \llbracket N \rrbracket \rangle$
\item $\llbracket \Gamma \VTS \PMV(M , (x , y).N) : C \rrbracket = \llbracket N \rrbracket \circ \alpha \circ \langle \ID , \llbracket M \rrbracket \rangle$
\item $\llbracket \Gamma \VTS \ast : \unit \rrbracket = \langle \rangle$
\end{enumerate}

Computation terms $\Gamma \CTS M : A$ are interpreted as morphisms $\llbracket \Gamma \rrbracket \xrightarrow{{\llbracket M \rrbracket}_S} S\llbracket A\rrbracket$.

\begin{enumerate}
  \item ${\llbracket \Gamma \CTS \RET (M) : X \rrbracket}_S = \eta \circ \llbracket M \rrbracket$
  \item ${\llbracket \Gamma \CTS \PUSH (W_1, x.W_2) : X\rrbracket}_S = {\llbracket W_2 \rrbracket}_S^{\sharp} \circ \STR \circ \langle \ID , {\llbracket W_1 \rrbracket}_S \rangle$
  \item ${\llbracket \Gamma \CTS \APP(M , N) : X \rrbracket}_S = \EVAL \circ \langle \llbracket M \rrbracket , \llbracket N \rrbracket \rangle$
  \item ${\llbracket \Gamma \CTS \VAR(L) : X \rrbracket}_S = \VAR_X \circ \llbracket L \rrbracket$
  \item ${\llbracket \Gamma \CTS \SUB(a.W_1 , W_2) : X \rrbracket}_S = \SUB_X \circ \langle \lambda {\llbracket W_1 \rrbracket}_S , {\llbracket W_2 \rrbracket}_S \rangle$
  \item ${\llbracket \Gamma \CTS \PMC(M , (x , y).W) : X \rrbracket}_S = {\llbracket W \rrbracket}_S \circ \alpha \circ \langle \ID , \llbracket M \rrbracket \rangle$
\end{enumerate}

CPS-translation of interpretation of computation terms $\Gamma \CTS M : A$ (i.e.\ as $\llbracket \Gamma \rrbracket \xrightarrow{\llbracket M \rrbracket} K_{\LABEL}\llbracket A\rrbracket$):
\begin{enumerate}
\item $\llbracket \Gamma \CTS \RET (M) : X \rrbracket \;\gamma\; k = k \;(\llbracket M \rrbracket\; \gamma)$
\item $\llbracket \Gamma \CTS \PUSH (W_1, x.W_2) : X\rrbracket \;\gamma \; k = ({\llbracket W_2 \rrbracket}^{\sharp} \circ \STR \circ \langle \gamma , \llbracket W_1 \rrbracket \;\gamma \rangle) \; k$ %\newline \hspace*{\fill} where ${\_}^{\sharp} : (X \to K_\LABEL Y) \to (K_\LABEL X \to K_\LABEL Y)$
\item $\llbracket \Gamma \CTS \APP(M , N) : X \rrbracket \; \gamma \; k = (\EVAL \circ \langle \llbracket M \rrbracket , \llbracket N \rrbracket \rangle) \; \gamma \; k$
\item $\llbracket \Gamma \CTS \VAR(L) : X \rrbracket \; \gamma \; k = \llbracket L \rrbracket \; \gamma$
\item $\llbracket \Gamma \CTS \SUB(a.W_1 , W_2) : X \rrbracket = \llbracket W_1 \rrbracket \; (\gamma , \llbracket W_2 \rrbracket \, \gamma \, k) \; k$
\item $\llbracket \Gamma \CTS \PMC(M , (x , y).W) : X \rrbracket \; \gamma \; k = (\llbracket W \rrbracket \circ \alpha \circ \langle \gamma , \llbracket M \rrbracket \rangle) k$
\end{enumerate}

Computation stacks $c$ for a context $\Gamma$ and type $X$ are interpreted as morphisms $K_{\LABEL}\llbracket X \rrbracket \xrightarrow{{\llbracket c \rrbracket}_C} K_{\LABEL}\llbracket R_0 \rrbracket$.
\begin{enumerate}
  \item ${\llbracket \square \rrbracket}_C = \ID$
  \item ${\llbracket W \triangleleft \gamma :: c \rrbracket}_C = {\llbracket c \rrbracket}_C \circ {\llbracket W \rrbracket}^{\sharp} \circ \STR \circ \langle \llbracket \gamma \rrbracket , \ID \rangle$
\end{enumerate}

Using the starting continuation $k_0 : \llbracket R_0 \rrbracket \to R$, we can interpret computation stacks $c$ for context $\Gamma$ and type $X$ as continuations $\llbracket X \rrbracket \to R$:
\[
  {\llbracket c \rrbracket} (x) = \big({\llbracket c \rrbracket}_C \; (\, \eta (x) \,) \big) \; k_0
\]

Environments $\gamma$ for context $\Gamma$ are interpreted as morphisms $1 \to \llbracket \Gamma \rrbracket$.
\begin{enumerate}
  \item $\llbracket () \rrbracket = \langle \rangle$
  \item $\llbracket (\gamma , M) \rrbracket = (\llbracket \gamma \rrbracket , \llbracket M \rrbracket \llbracket \gamma \rrbracket)$
  \item $\llbracket (\gamma , (W ; c)) \rrbracket = (\llbracket \gamma \rrbracket , \llbracket W \rrbracket \llbracket \gamma \rrbracket \llbracket c \rrbracket)$
\end{enumerate}

\subsection{Lookup Machine}
\subsubsection{States}$ $

A lookup state for type $X$ is a pair $\langle x \,|\, \gamma \rangle$, where $x$ is a variable in $\Gamma$ (i.e.\ an index into $\Gamma$) and $\gamma$ is an environment for $\Gamma$.

The halting states are: $\langle 0 \,|\, (\gamma , \underline{\ast}) \rangle$, $\langle 0 \,|\, (\gamma , \underline{\lambda} x . W) \rangle$, $\langle 0 \,|\, (\gamma , (\underline{M}, \underline{N})) \rangle$ and $\langle 0 \,|\, (\gamma , (W ; c)) \rangle$

\subsubsection{Transitions}

\begin{align*}
  \langle 0 \,|\, (\gamma , \underline{\ell}) \rangle &\leadsto^L \langle \underline{\ell} \,|\, \gamma \rangle && \text{where $\ell$ is a label variable} \\
  \langle x + 1 \,|\, (\gamma , \underline{M}) \rangle &\leadsto^L \langle x \,|\, \gamma \rangle\\
  \langle x + 1 \,|\, (\gamma , (W ; c)) \rangle &\leadsto^L \langle x \,|\, \gamma \rangle
\end{align*}


\subsection{Value Machine}
\subsubsection{States}
Value states are associated with a type $X$.

Value states are stacks, which are inductively defined as follows.
\begin{itemize}
  \item The empty value stack of type $X$ is denoted by $\square_X$ (or just $\square$).
  \item If $M$ is a partial value term of type $X$ in context $\Gamma$ and $\gamma$ is an environment for $\Gamma$, then $M \triangleleft \gamma :: \square_X$ is a value stack of type $X$.
  \item If $v$ is a non-empty value stack of type $X$ and $M$ is a partial value term in context $\Gamma$ (and of any type) and $\gamma$ is an environment for $\Gamma$, then $M \triangleleft \gamma :: v$ is a value stack of type $X$.
\end{itemize}

Halting states are those of the form $\underline{M} \triangleleft \gamma :: \square$ (i.e.\ those with only one evaluated value term on the stack).

\subsubsection{Transitions}

\begin{align*}
  \ell \triangleleft \gamma :: v &\leadsto^V \underline{l} \triangleleft \gamma :: v && \text{where $\ell$ is a label variable} \\
  x \triangleleft \gamma :: v &\leadsto^V \underline{M} \triangleleft \gamma :: v && \text{if $\langle \ell \,|\, \gamma \rangle \leadsto^{L*} \langle 0 \,|\, (\gamma' , \underline{M}) \rangle$} \\
  \lambda x . W \triangleleft \gamma :: v &\leadsto^V \underline{\lambda} x . W \triangleleft \gamma :: v\\
  (M , N) \triangleleft \gamma :: v &\leadsto^V M \triangleleft \gamma :: (\_ , N) \triangleleft \gamma :: v\\
  \PMV (M , (x,y).N) \triangleleft \gamma :: v &\leadsto^V M \triangleleft \gamma :: \PMV(\_ , N) \triangleleft \gamma :: v\\
  \ast\, \triangleleft\, \gamma :: v &\leadsto^V \underline{\ast} \triangleleft \gamma :: v\\
  \underline{M} \triangleleft \gamma :: (\_ , N) \triangleleft \gamma' :: v &\leadsto^V N \triangleleft \gamma :: (\underline{M} , \_) \triangleleft \gamma :: v\\
  (\underline{M}, \underline{N}) \triangleleft \gamma :: \PMV(\_ , N) \triangleleft \gamma' :: v &\leadsto^V N \triangleleft (\gamma , \underline{M}, \underline{N}) :: v\\
\end{align*}

\subsection{Computation Machine}

\subsubsection{States}

A computation state is a triple of a computation term $\Gamma, \Gamma' \CTS W : X$ or $\Gamma, \Gamma' \CTS \underline{W} : X$ (i.e.\ a full computation term or an evaluated one), an environment $\gamma$ for $\Gamma, \Gamma'$ and a computation stack $c$ for context $\Gamma$ and type $X$.
%
We denote these triples by $\langle W \prec \gamma \,|\, c \rangle$.
%
The only halting state is $\langle \underline{\RET} (\underline{M}) \prec \gamma \,|\, \square \rangle$.
%
We assume that there is a starting continuation $k_0 : \llbracket R_0 \rrbracket \to R$.

\subsubsection{Transitions}

\begin{align*}
  \langle \RET (M) \prec \gamma \,|\, c \rangle &\leadsto^C \langle \underline{\RET} (\underline{M'}) \prec \gamma' \,|\, c \rangle\\ & \hspace{120pt} \text{if $M \triangleleft \gamma :: \square \leadsto^{V*} \underline{M'} \triangleleft \gamma' :: \square$} \\
  \langle \underline{\RET} (\underline{M}) \prec \gamma \,|\, N \triangleleft \gamma' :: c \rangle &\leadsto^C \langle N \prec (\gamma, \underline{M}) \,|\, c \rangle \\
  \langle \PUSH (M, x.N) \prec \gamma \,|\, c \rangle &\leadsto^C \langle M \prec \gamma \,|\, N \triangleleft \gamma :: c \rangle \\
  \langle \SUB (a.W_1, W_2) \prec \gamma \,|\, c \rangle &\leadsto^C \langle M \prec (\gamma , (N ; c) \,|\, c \rangle \\
  \langle \PMC (M, (x, y).W) \prec \gamma \,|\, c \rangle &\leadsto^C \langle W \prec (\gamma' , M_1 , M_2) \,|\, c \rangle\\ & \hspace{120pt} \text{if $M \triangleleft \gamma :: \square \leadsto^{V*} (\underline{M_1}, \underline{M_2}) \triangleleft \gamma' :: \square$}  \\
  \langle \APP (M , N) \prec \gamma \,|\, c \rangle &\leadsto^C \langle \underline{\APP} (M , \underline{N'}) \prec \gamma' \,|\, c \rangle\\ & \hspace{120pt} \text{if $N \triangleleft \gamma :: \square \leadsto^{V*} \underline{N'} \triangleleft \gamma' :: \square$} \\
  \langle \underline{\APP} (x , \underline{N}) \prec \gamma \,|\, c \rangle &\leadsto^C \langle \underline{\APP} (\lambda x.W , \underline{N}) \prec \gamma \,|\, c \rangle\\ & \hspace{120pt} \text{if $\langle x \,|\, \gamma \rangle \leadsto^{L*} \langle \underline{\lambda} x.W \,|\, \gamma' \rangle$}  \\
  \langle \underline{\APP} (\PMV (M, N_1) , \underline{N_2}) \prec \gamma \,|\, c \rangle &\leadsto^C \langle \underline{\APP} (N_1 , \underline{N}) \prec (\gamma' , \underline{M_1} , \underline{M_2}) \,|\, c \rangle\\ & \hspace{120pt} \text{if $M \triangleleft \gamma :: \square \leadsto^{V*} (\underline{M_1} , \underline{M_2}) \triangleleft \gamma' :: \square$}  \\
  \langle \underline{\APP} (\lambda x. W , \underline{N}) \prec \gamma \,|\, c \rangle &\leadsto^C \langle W \prec (\gamma , \underline{N}) \,|\, c \rangle  \\
  \langle \VAR (M) \prec \gamma \,|\, c \rangle &\leadsto^C \langle W \prec \gamma' \,|\, c' \rangle \\
    & \hspace{120pt} \text{if $M \triangleleft \gamma :: \square \leadsto^{V*} \underline{\ell} \triangleleft \gamma' :: \square$} \\
    & \hspace{120pt} \text{and $\langle \ell \,|\, \gamma' \rangle \leadsto^{L*} \langle 0 \,|\, (\gamma'', (W ; c')) \rangle$} \\
\end{align*}
