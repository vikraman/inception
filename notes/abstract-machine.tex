\section{Abstract Machine}

\subsection{Terms}
We have value terms and computation terms (subject to term formation rules, which are provided elsewhere).
For the purposes of the abstract machine, we distinguish between full terms and partial terms.
Every full term is also a partial term.

Full value terms are:
$x$,
$\ast$,
$\lambda x . M$,
$(L , R)$,
$\PM ( (a , b) =  M , N )$
\newline
Full computation terms are:
$\RET (M)$,
$\PUSH (M , x.N)$,
$\APP (M , N)$,
$\VAR (M)$,
$\SUB (a.M, N)$
\newline
Further partial value terms are:
$(\_ , R)$,
$(L , \_)$,
$\flag{(L , R)}$
\newline
Further partial computation terms are:
$\APP(\_ , N)$,
$\RET (\_)$,
$\flag{\RET (M)}$

\subsection{States}

States are pairs $(t ; k)$, where $t$ is a stack of partial terms in an environment (a pair of a partial term and an environment) and $k$ is a stack of continuations.
If the partial term at the top of the stack $t$ is a partial value term, then $(t ; k)$ is a \emph{value state}, otherwise $(t ; k)$ is a \emph{computation state}.

An \emph{environment} is a map from variables to computation states and full terms in an environment.

\subsection{Transitions}
\begin{align*}
	%
	\intertext{Variable lookup:}
	((x, \gamma), \tail ; k) &\leadsto \left\{
	\begin{array}{ll}
		((\pi_1 \gamma (x) , \pi_2 \gamma (x)) , \tail ; \pi_3 \gamma (x)) & \text{if $\gamma (x)$ is a c.s.} \\
		((\pi_1 \gamma (x) , \pi_2 \gamma (x)) , \tail ; k) & \text{otherwise}
	\end{array}
	\right.\\
	\\
  %
	\intertext{From value states:}
  (( \lambda x . M , \gamma) , ( \APP(\_ , N) , \gamma') , \tail ; k ) &\leadsto (( M , \gamma [ x \mapsto N ]) , \tail ; k)\\
  (( (L , R) , \gamma ) , \tail ; k ) &\leadsto (( L , \gamma ) , ( (\_ , R) , \gamma) , \tail ; k)\\
  (( \flag{(L, R)}, \gamma), (\PM ( (a , b) = \_ , N) , \gamma') , \tail ; k) &\leadsto (( N , \gamma' [ a \mapsto (L , \gamma) , b \mapsto (R , \gamma) ]) , \tail ; k )\\
  (( \PM (L , R), \gamma) , \tail ; k) &\leadsto (( M , \gamma) , (\PM ((a , b) = \_ , N) , \gamma) , \tail ; k )\\
	\\
	%
  (( T , \gamma) , ( (\_ , R) , \gamma') , \tail ; k ) &\leadsto (( R , \gamma' ) , (T , \_) , \tail ; k )\\
  (( T , \gamma) , ( (L , \_) , \gamma') , \tail ; k ) &\leadsto (\flag{(L , T)} , \tail ; k )\\
  (( T , \gamma ) , (\RET (\_) , \gamma') , \tail ; k ) &\leadsto (( \RET (T) , \gamma') , \tail ; k)\\
	\intertext{\hspace*{\fill}where $T = \lambda x . M$ or $T = \flag{(L , R)}$ or $T = \ast$}
  %
	% computations
	%
	\\
	\intertext{From computation states:}
  (( \RET (M) , \gamma) , \tail ; k ) &\leadsto ( (M , \gamma) , (\RET (\_) , \gamma) , \tail ; k)\\
  (( \flag{\RET (\_)}, \gamma) , \tail ; x.N , k) &\leadsto (( N , \gamma[ x \mapsto M]) , \tail ; k)\\
  (( \PUSH (M , x.N) , \gamma) , \tail ; k ) &\leadsto (( M , \gamma ) , \tail ; x.N , k )\\
  (( \APP (M , N) , \gamma ) , \tail ; k ) &\leadsto (( M , \gamma) , (\APP (\_ , N) , \gamma) , \tail ; k )\\
  (( \VAR (M) , \gamma) , \tail ; k ) &\leadsto (( M , \gamma ) ; k )\\
  (( \SUB (a.M , N ), \gamma) , \tail ; k ) &\leadsto (( M , \gamma [a \mapsto (N , \gamma, k)] ) , \tail ; k )
\end{align*}
