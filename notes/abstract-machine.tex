\subsection{Contexts}

\subsubsection{Grammar}

\begin{align*}
  \text{Types:} && X \,,\, Y \,,\, Z  &:=\; \unit \mid X \Rightarrow Y \mid X \times Y \mid \LABEL\\
  \text{Contexts:} && \Gamma \,,\, \Delta &:=\; \emptyctx \mid \Gamma , X\\
  \text{Variables:} && \ell \,,\, x \,,\, y &:=\; h \mid t ( x )
\end{align*}

The letter $\ell$ will be used to denote variables of type $\LABEL$ exclusively, while the letters $x$ and $y$ will be used for all variables.

Furthermore, we let $R_0 = V$ (this is a type) and $R = \VP$ (this is a presheaf) for the rest of the document.
We also fix a continuation $k_0 : \llbracket R_0 \rrbracket \to \llbracket R \rrbracket$ (interpretation is given below).

\subsubsection{Typing rules for membership}
\begin{align*}
   \frac{}{h : X \in \Gamma , X}
  && \frac{x : X \in \Gamma}{t(x) : X \in \Gamma , Y}
\end{align*}

\subsubsection{Interpretation of types for monad T (where $T = S$ or $T = K_R$)}$ $

Types are interpreted as presheaves in $\textsf{Set}^\mathbb{F}$.
\begin{align*}
  {\llbracket \unit \rrbracket}_T &= \top\\
  {\llbracket X \times Y \rrbracket}_T &= {\llbracket X \rrbracket}_T \times {\llbracket Y \rrbracket}_T\\
  {\llbracket X \Rightarrow Y \rrbracket}_T &= {\llbracket X \rrbracket}_T \Rightarrow T{\llbracket Y \rrbracket}_T\\
  {\llbracket \LABEL \rrbracket}_T &= R
\end{align*}

If $T = K_R$, we will just write $\llbracket X \rrbracket$ instead of ${\llbracket X \rrbracket}_{K_R}$.\newline

\subsubsection{Interpretation of contexts for monad T (where $T = S$ or $T = K_R$)}

\begin{align*}
  {\llbracket \emptyctx \rrbracket}_T &= \top\\
  {\llbracket \Gamma, X \rrbracket}_T &= {\llbracket \Gamma \rrbracket}_T \times {\llbracket X \rrbracket}_T
\end{align*}

If $T = K_R$, we will just write $\llbracket \Gamma \rrbracket$ instead of ${\llbracket \Gamma \rrbracket}_{K_R}$.\newline

\subsubsection{Interpretation of variables}$ $

Variables $x : X \in \Gamma$ are interpreted as morphisms $\llbracket \Gamma \rrbracket \xrightarrow{\llbracket x \rrbracket} \llbracket X \rrbracket$ in $\textsf{Set}^\mathbb{F}$.

\begin{align*}
  {\llbracket h \rrbracket} &= \pi_2\\
  {\llbracket t(x) \rrbracket} &= {\llbracket x \rrbracket}_T \circ \pi_1
\end{align*}

\subsection{Terms}

\subsubsection{Grammar}

\begin{align*}
  \text{Full value terms:} && M \,,\, N &:=\; x \mid \ast \mid \lambda x . W \mid (M , N) \mid \PMV ( M , (x,y).N )\\
  \text{Evaluated value terms:} && \underline{M} \,,\, \underline{N} &:=\; \underline{\ast} \mid (\underline{M} \mid \underline{N}) \mid \underline{\lambda} x . W \mid \underline{\ell} \;\text{(where $\ell$ is a label variable)}\\
  \text{Partial value terms:} && \overline{M} \,,\, \overline{N} &:=\; (-_M , N) \mid (\underline{M} , -_N) \mid \PMV(-_M, N)\\
  \text{Value terms:} && \mathbf{M} \,,\, \mathbf{N} &:=\; M \mid \underline{M} \mid \overline{M}\\
  \text{Full computation terms:} && W &:=\; \RET (M) \mid \PUSH (M , N) \mid \APP (M , N) \mid \\
                                          &&& \quad\quad \VAR (M) \mid \SUB (M, N) \mid \PMC ( M , N )\\
  \text{(Partially) evaluated}&\\
  \text{computation terms:} && \underline{W} &:=\; \underline{\APP} (M , \underline{N}) \mid \underline{\RET} (\underline{M})\\
  \text{Computation terms:} && \mathbf{W} &:=\; W \mid \underline{W}
\end{align*}


\subsubsection{Typing rules for full value terms}$ $

\begin{align*}
   \frac{x : X \in \Gamma}{\Gamma \VTS x : X}
&& \frac{\Gamma , X \CTS M : Y}{\Gamma \VTS \lambda M : X \Rightarrow Y}
&& \frac{\Gamma \VTS M : X \quad \Gamma \VTS N : Y}{\Gamma \VTS (M , N) : X \times Y}
\end{align*}
\begin{align*}
\frac{\Gamma \VTS M : X \times Y \quad \Gamma, x : X , y : Y \VTS N : Z}{\Gamma \VTS \PMV(M, N) : Z}
&& \frac{}{\Gamma \VTS \ast : \unit}
\end{align*}

\subsubsection{Typing rules for evaluated value terms}$ $

\begin{align*}
  \frac{}{\Gamma \VVTS \underline{\ast} : \unit}
  && \frac{\Gamma \VVTS \underline{M} : X \quad \Gamma \VVTS \underline{N} : Y}{\Gamma \VVTS (\underline{M} , \underline{N}) : X \times Y}
  &&\frac{\ell : \LABEL \in \Gamma}{\Gamma \VVTS \underline{\ell} : \LABEL}
  && \frac{\Gamma , X \CTS W : Y}{\Gamma \VVTS \underline{\lambda} W : X \Rightarrow Y}
\end{align*}

\subsubsection{Typing rules for partial value terms}$ $

\begin{align*}
  \frac{\Gamma \VTS M : X \quad \Gamma \VTS N : Y}{\Gamma \VPTS (-_M , N) : X \times Y}
  && \frac{\Gamma \VVTS \underline{M} : X \quad \Gamma \VTS N : Y}{\Gamma \VPTS (\underline{M} , -_N) : X \times Y}
  && \frac{\Gamma \VTS M : X \times Y \quad \Gamma, x : X , y : Y \VTS N : Z}{\Gamma \VPTS \PMV(-_M, N) : Z}
\end{align*}

\subsubsection{Typing rules for full computation terms}$ $

\begin{align*}
   \frac{\Gamma \VTS M : X}{\Gamma \CTS \RET (M) : X}
&& \frac{\Gamma \CTS W : X \quad \Gamma, X \CTS Y}{\Gamma \CTS \PUSH (M , N) : Y}
&& \frac{\Gamma \VTS M : X \Rightarrow Y \quad \Gamma \VTS N : X}{\Gamma \CTS \APP(M, N) : Y}
\end{align*}
\begin{align*}
   \frac{\Gamma \VTS M : \LABEL}{\Gamma \CTS \VAR_X (M) : X}
&& \frac{\Gamma , a : \LABEL \CTS M : X \quad \Gamma \CTS N : X}{\Gamma \CTS \SUB (a.M , N) : X}
&& \frac{\Gamma \VTS M : X \times Y \quad \Gamma, x : X , y : Y \CTS N : Z}{\Gamma \CTS \PMC(M, (x,y).N) : Z}
\end{align*}

\subsubsection{Typing rules for (partially) evaluated computation terms}$ $

\begin{align*}
  \frac{\Gamma \VTS M : X}{\Gamma \CCTS \underline{\RET} (\underline{M}) : X}
  && \frac{\Gamma \VTS M : X \Rightarrow Y \quad \Gamma \VVTS \underline{N} : X}{\Gamma \CCTS \underline{\APP}(M, \underline{N}) : Y}
\end{align*}

For the judgements of value terms $\mathbf{M}$ and computation terms $\mathbf{W}$ we will use the notations $\Gamma \VTS \mathbf{M} : X$ and $\Gamma \CTS \mathbf{W} : X$, respectively, which are defined in the obvious way.

\subsubsection{Interpretation of full value terms for monad T (where $T = S$ or $T = K_R$)}$ $

Full value terms $\Gamma \VTS M : X$ are interpreted as morphisms ${\llbracket \Gamma \rrbracket}_T \xrightarrow{{\llbracket M \rrbracket}_T} {\llbracket X \rrbracket}_T$.

\begin{align*}
  {\llbracket \Gamma \VTS x : X \rrbracket}_T &= {\llbracket x \rrbracket}_T \\
  {\llbracket \Gamma \VTS \lambda W : X \Rightarrow Y \rrbracket}_T &= \lambda {\llbracket W \rrbracket}_T \\
  {\llbracket \Gamma \VTS (M, N) : X \times Y \rrbracket}_T &= \langle {\llbracket M \rrbracket}_T, {\llbracket N \rrbracket}_T \rangle \\
  {\llbracket \Gamma \VTS \PMV(M , N) : X \rrbracket}_T &= {\llbracket N \rrbracket}_T \circ \alpha \circ \langle \ID , {\llbracket M \rrbracket}_T \rangle \\
  {\llbracket \Gamma \VTS \ast : \unit \rrbracket}_T &= \top
\end{align*}

If $T = K_R$, we will just write $\llbracket \Gamma \VTS M : X \rrbracket$ instead of ${\llbracket \Gamma \VTS M : X \rrbracket}_{K_R}$.\newline


\subsubsection{Interpretation of full computation terms for monad $S$}$ $

Computation terms $\Gamma \CTS W : X$ are interpreted by ${\llbracket \_ \rrbracket}_S$ as morphisms ${\llbracket \Gamma \rrbracket}_S \xrightarrow{{\llbracket W \rrbracket}_S} S\llbracket X\rrbracket$.

\begin{align*}
  {\llbracket \Gamma \CTS \RET (M) : X \rrbracket}_S &= \eta \circ {\llbracket M \rrbracket}_S \\
  {\llbracket \Gamma \CTS \PUSH (W_1, x.W_2) : X\rrbracket}_S &= {\llbracket W_2 \rrbracket}_S^{\sharp} \circ \STR \circ \langle \ID , {\llbracket W_1 \rrbracket}_S \rangle\\
  {\llbracket \Gamma \CTS \APP(M , N) : X \rrbracket}_S &= \EVAL \circ \langle {\llbracket M \rrbracket}_S , {\llbracket N \rrbracket}_S \rangle\\
  {\llbracket \Gamma \CTS \VAR(M) : X \rrbracket}_S &= \VAR_X \circ {\llbracket M \rrbracket}_S\\
  {\llbracket \Gamma \CTS \SUB(a.W_1 , W_2) : X \rrbracket}_S &= \SUB_X \circ \langle \lambda {\llbracket W_1 \rrbracket}_S , {\llbracket W_2 \rrbracket}_S \rangle\\
  {\llbracket \Gamma \CTS \PMC(M , (x , y).W) : X \rrbracket}_S &= {\llbracket W \rrbracket}_S \circ \alpha \circ \langle \ID , \llbracket M \rrbracket \rangle
\end{align*}

\subsubsection{Interpretation of full computation terms for monad $K_R$}$ $

Computation terms $\Gamma \CTS W : X$ are interpreted by $\llbracket \_ \rrbracket$ as morphisms $\llbracket \Gamma \rrbracket \xrightarrow{\llbracket W \rrbracket} K_R\llbracket X\rrbracket$.

\begin{align*}
  \llbracket \Gamma \CTS \RET (M) : X \rrbracket \;\gamma\; k &= k \;(\llbracket M \rrbracket\; \gamma)\\
  \llbracket \Gamma \CTS \PUSH (W_1, x.W_2) : X\rrbracket \;\gamma \; k &= ({\llbracket W_2 \rrbracket}^{\sharp} \circ \STR \circ \langle \gamma , \llbracket W_1 \rrbracket \;\gamma \rangle) \; k\\
  \llbracket \Gamma \CTS \APP(M , N) : X \rrbracket \; \gamma \; k &= (\EVAL \circ \langle \llbracket M \rrbracket , \llbracket N \rrbracket \rangle) \; \gamma \; k\\
  \llbracket \Gamma \CTS \VAR(L) : X \rrbracket \; \gamma \; k &= \llbracket L \rrbracket \; \gamma\\
  \llbracket \Gamma \CTS \SUB(a.W_1 , W_2) : X \rrbracket &= \llbracket W_1 \rrbracket \; (\gamma , \llbracket W_2 \rrbracket \, \gamma \, k) \; k\\
  \llbracket \Gamma \CTS \PMC(M , (x , y).W) : X \rrbracket \; \gamma \; k &= (\llbracket W \rrbracket \circ \alpha \circ \langle \gamma , \llbracket M \rrbracket \rangle) k
\end{align*}

\subsection{Environments}

\subsubsection{Grammar}

\begin{align*}
  \text{Computation Stack:} && C &:=\; \emptycs \mid \cstack{W}{\gamma}{C}\\
  \text{Environment:} && \gamma &:=\; \emptyenv \mid \gamma , M \mid \gamma , \cspair{W}{C}
\end{align*}

\subsubsection{Typing rules for computation stacks}

\begin{align*}
  \frac{}{\emptycs : \CStack{\emptyctx}{R_0}}
  && \frac{\Gamma , Z \CTS M : X \quad \gamma : \Env{\Gamma} \quad C : \CStack{\Delta}{X} \quad \Gamma = \Delta , \Delta'}{\cstack{M}{\gamma}{C} : \CStack{\Gamma}{Z}}
\end{align*}

\subsubsection{Typing rules for environments}
\begin{align*}
  \frac{}{\emptyenv : \Env{\emptyctx}}
  && \frac{\gamma : \Env{\Gamma} \quad \Gamma \VVTS \underline{M} : X }{\gamma , \underline{M} : \Env{\Gamma , X}}
  && \frac{\gamma : \Env{\Gamma} \quad \Gamma \CTS W : X \quad C : \CStack{\Delta}{X} \quad \Gamma = \Delta, \Delta'}{\gamma , \cspair{W}{C} : \Env{\Gamma , \LABEL}}
\end{align*}

\subsubsection{Interpretation of computation stacks}$ $

Computation stacks for any context and type $X$ are interpreted as continuations $\llbracket X \rrbracket \to R$.

In order to define this interpretation inductively, we first set up an intermediate interpretation ${\llbracket \_ \rrbracket}_K$, mapping computation stacks $C$ for any context and type $X$ to morphisms $K_R\llbracket X \rrbracket \xrightarrow{{\llbracket C \rrbracket}_K} K_{\LABEL}\llbracket R_0 \rrbracket$.
%
\begin{align*}
  {\llbracket \emptycs \rrbracket}_K &= \ID\\
  {\llbracket W \triangleleft \gamma :: C \rrbracket}_K &= {\llbracket C \rrbracket}_K \circ {\llbracket W \rrbracket}^{\sharp} \circ \STR \circ \langle \llbracket \gamma \rrbracket , \ID \rangle
\end{align*}

Using the starting continuation $k_0 : \llbracket R_0 \rrbracket \to R$, we can now interpret computation stacks $C$ for any context and for type $X$ as continuations $\llbracket X \rrbracket \to R$:
\[
  {\llbracket C \rrbracket} (x) = \big({\llbracket C \rrbracket}_K \; (\, \eta (x) \,) \big) \; k_0
\]

\subsubsection{Interpretation of environments}$ $

Environments $\gamma$ for context $\Gamma$ are interpreted as morphisms $\top \to \llbracket \Gamma \rrbracket$.
\begin{enumerate}
  \item $\llbracket () \rrbracket = \langle \rangle$
  \item $\llbracket (\gamma , M) \rrbracket = (\llbracket \gamma \rrbracket , \llbracket M \rrbracket \llbracket \gamma \rrbracket)$
  \item $\llbracket (\gamma , (W ; c)) \rrbracket = (\llbracket \gamma \rrbracket , \llbracket W \rrbracket \llbracket \gamma \rrbracket \llbracket c \rrbracket)$
\end{enumerate}
