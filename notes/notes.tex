\section{Substitution Algebras}

\begin{definition}[Substitution Algebra]
    \label{def:substitution-algebra}
    A $V$-\emph{substitution algebra} in a category $\cat{C}$ for $V \in \cat{C}$ is an object $X$ with maps:
    \begin{align*}
        \var\colon & V \to X            \\
        \sub\colon & X^V \times X \to X \\
    \end{align*}
    satisfying:

        \item \emph{Weakening}:

                      \arrow["{X^{!_V} \times 1}"', from=1-1, to=3-1]
                      \arrow["{\pi_1}", from=1-3, to=3-3]
                      \arrow["\sub"{description}, from=3-1, to=3-3]
                  \end{tikzcd}\]

        \item \emph{Substitution}:
